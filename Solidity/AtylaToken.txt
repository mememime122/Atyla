pragma solidity 0.8.0;

import "https://github.com/OpenZeppelin/openzeppelin-contracts/blob/v4.3.2/contracts/token/ERC20/ERC20.sol";
import "https://github.com/OpenZeppelin/openzeppelin-contracts/blob/v4.3.2/contracts/access/Ownable.sol";

contract AtylaToken is ERC20, Ownable{
    constructor() ERC20("Atyla Token", "ATYLA"){
        //transferOwnership("address de contrato RifasViajes");
    }

    function mint(uint256 amount) external virtual onlyOwner{
         _mint(msg.sender, amount *10**uint(decimals()-2));
    }

    function burn(address account, uint256 amount) external virtual onlyOwner{
        _burn(account, amount *10**uint(decimals()-2));
    }

}
